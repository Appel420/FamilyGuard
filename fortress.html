use blake3::Hasher;
use argon2::{Argon2, PasswordHasher, password_hash::{SaltString, PasswordHash, PasswordVerifier, rand_core::OsRng}};
use aes_gcm::{Aes256Gcm, Key, Nonce, KeyInit};
use aes_gcm::aead::{Aead, OsRng as AesOsRng};
use std::fs::{File, OpenOptions};
use std::io::{Write, Read};

// 1. Generate a cryptographically random salt (hardware entropy via OsRng)
fn gen_salt() -> SaltString {
    SaltString::generate(&mut OsRng)
}

// 2. Hash with Argon2id (memory-hard, side-channel resistant)
fn slow_hash(password: &str, salt: &SaltString) -> Result<String, Box<dyn std::error::Error>> {
    let argon2 = Argon2::new(
        argon2::Algorithm::Argon2id,
        argon2::Version::V0x13,
        argon2::Params::default(),
    );
    let hash = argon2.hash_password(password.as_bytes(), salt)?.to_string();
    Ok(hash)
}

// 3. Blake3 checksum – fastest, cryptographically secure hash
fn checksum(data: &[u8]) -> [u8; 32] {
    *Hasher::new().update(data).finalize().as_bytes()
}

// 4. Derive a 256-bit AES key from the Argon2id hash output
fn derive_aes_key(hash_str: &str) -> Key<Aes256Gcm> {
    let raw = checksum(hash_str.as_bytes());
    *Key::<Aes256Gcm>::from_slice(&raw)
}

// 5. Compress with Zstd before encryption (reduces ciphertext size)
fn compress_data(data: &[u8]) -> Vec<u8> {
    zstd::encode_all(data, 9).unwrap_or_else(|_| data.to_vec())
}

fn decompress_data(data: &[u8]) -> Vec<u8> {
    zstd::decode_all(data).unwrap_or_else(|_| data.to_vec())
}

// 6. MAIN: Lock a file with Fortress Protocol
//    Output format: <12-byte nonce> | <AES-GCM ciphertext> | <blake3 checksum> | <argon2 salt>
pub fn fortress_lock(path: &str, password: &str) -> Result<(), Box<dyn std::error::Error>> {
    let mut file = File::open(path)?;
    let mut data = Vec::new();
    file.read_to_end(&mut data)?;

    let salt = gen_salt();
    let hash = slow_hash(password, &salt)?;
    let key = derive_aes_key(&hash);
    let cipher = Aes256Gcm::new(&key);

    let nonce_bytes = Aes256Gcm::generate_nonce(&mut AesOsRng);
    let nonce = Nonce::from_slice(&nonce_bytes);

    let compressed = compress_data(&data);
    let sealed = cipher.encrypt(nonce, compressed.as_ref())
        .map_err(|e| format!("Encryption failed: {}", e))?;

    let checksummed = checksum(&sealed);

    let mut out = OpenOptions::new().create(true).write(true).truncate(true)
        .open(format!("{}.fortress", path))?;
    out.write_all(&nonce_bytes)?;
    out.write_all(b"|")?;
    out.write_all(&sealed)?;
    out.write_all(b"|")?;
    out.write_all(&checksummed)?;
    out.write_all(b"|")?;
    out.write_all(salt.as_str().as_bytes())?;
    out.flush()?;

    println!("🔒 Fortress locked: {}.fortress", path);
    println!("  → Argon2id: slow hashed");
    println!("  → AES-256-GCM: authenticated encryption");
    println!("  → Blake3: integrity verified");
    println!("  → Zstd: compressed");
    Ok(())
}

// 7. MAIN: Open a Fortress-locked file
pub fn fortress_open(path: &str, password: &str) -> Result<Vec<u8>, Box<dyn std::error::Error>> {
    let mut file = File::open(path)?;
    let mut contents = Vec::new();
    file.read_to_end(&mut contents)?;

    // Format: <12-byte nonce> | <ciphertext> | <blake3 checksum> | <argon2 salt>
    let parts: Vec<&[u8]> = contents.splitn(4, |&b| b == b'|').collect();
    if parts.len() != 4 {
        return Err("Invalid fortress file format".into());
    }

    let nonce_bytes = parts[0];
    let sealed = parts[1];
    let stored_checksum = parts[2];
    let salt_str = std::str::from_utf8(parts[3])?;
    let salt = SaltString::from_b64(salt_str).map_err(|e| e.to_string())?;

    // Verify integrity before decryption
    let live_checksum = checksum(sealed);
    if live_checksum != stored_checksum {
        return Err("Checksum mismatch — file may be tampered".into());
    }

    let hash = slow_hash(password, &salt)?;
    let key = derive_aes_key(&hash);
    let cipher = Aes256Gcm::new(&key);
    let nonce = Nonce::from_slice(nonce_bytes);

    let decrypted = cipher.decrypt(nonce, sealed)
        .map_err(|_| "Decryption failed — wrong password or corrupted file")?;

    Ok(decompress_data(&decrypted))
}
